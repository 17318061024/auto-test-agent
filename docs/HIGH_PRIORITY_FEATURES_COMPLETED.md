```typescript
// 智能检测策略
1. 系统Chrome (用户安装的)
2. 内置Chromium (项目打包的)
3. Playwright Chromium (自动下载的)

// 健康检查
const health = await chromeManager.healthCheck(executablePath)
if (!health.healthy) {
```
